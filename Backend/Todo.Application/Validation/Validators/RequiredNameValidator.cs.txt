using Todo.Application.Exceptions;

namespace Todo.Application.Validation.Validators;

public class RequiredNameValidator : ITaskValidator
{
    public Task ValidateAsync(ValidationContext context)
    {
        var name = context.IsCreate
            ? context.CreateRequest!.Name
            : context.UpdateRequest!.Name;

        if (string.IsNullOrWhiteSpace(name))
            throw new ValidationException("Task name is required.");

        return Task.CompletedTask;
    }
}
